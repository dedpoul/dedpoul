1)git init                            -Установка гита
2)git config --global user.name ""
3)git config --global user.email ""   -Регистрируемся
4)git add .                           -Добавляем все файлы
5)git status                          -Проверяем что мы добавляем
6)git commit  -m 'комментарий'        -Подтверждаем

7)git branch vetka                    -Создаём новую ветку
8)git checkout vetka                  -Переходим на новую ветку
9)git status                          -Проверяем

//Далее по заданию в новой ветке создаём папку blog с файлами, и добавляем её в эту ветку
10)git add .                           -Добавляем все файлы
11)git status                          -Проверяем что мы добавляем
12)git commit  -m 'комментарий'        -Подтверждаем
13)git checkout master                  -Переходим на стандартную(основую) ветку и проверяем наличие папки(её быть не должно)
